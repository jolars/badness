% OpenGraph card: 1200 x 630 pt (the de-facto social-preview size).
%
% Page units are pt-as-pixels, so the resulting PDF is exactly 1200x630 pt;
% it scales cleanly to any integer multiple at higher dpi.
%
% Layout is plain boxes: a fixed-size minipage with \hfill on both sides,
% so the mark + text cluster is truly centered. Only the mark is tikz.
\documentclass[border=0pt]{standalone}

% ============================================================================
% KNOBS — edit freely
% ============================================================================

% --- Mark size (shape is canonical in _common.tex) ---
\providecommand{\bnMarkScale}{1.8}

% --- Layout (pt) ---
\providecommand{\bnMarkGap}{85}
% horizontal gap between mark and text
\providecommand{\bnMarkYNudge}{0}
% raise (+) or lower (-) the mark
\providecommand{\bnSubtitleSkip}{42}  % extra space between title and subtitle

\providecommand{\bnTitleSize}{120} % title font size (pt)
\providecommand{\bnSubtitleSize}{28}  % subtitle font size (pt)

% --- Text content ---
\providecommand{\bnTitle}{Badness}
\providecommand{\bnShowSubtitle}{1} % 0 = title only, 1 = title + subtitle
\providecommand{\bnSubtitle}{A \LaTeX{} formatter, linter \& language server}

% --- Theme ---
\providecommand{\bnDarkMode}{0}
\providecommand{\bnFont}{fira}

\input{_common.tex}

\pagecolor{bnPaper}

% ============================================================================
\begin{document}
\begin{minipage}[c][630pt][c]{1200pt}%
  \hfill
  % The mark, vertically centered on the line.
  \raisebox{\bnMarkYNudge pt}{%
    \begin{tikzpicture}[baseline=(current bounding box.center)]
      \bnDrawMark
    \end{tikzpicture}
  }
  %
  \hspace{\bnMarkGap pt}
  %
  % Text column: a one-column tabular, vertically centered on the line.
  \begin{tabular}{@{}l@{}}{
    \fontsize{\bnTitleSize pt}{\bnTitleSize pt}\bfseries\selectfont
    \textcolor{bnInk}{\bnTitle}
  }%
    \ifnum\bnShowSubtitle=1 \\[\bnSubtitleSkip pt]
    \hspace{4pt}% optical alignment with the title's left edge
    {
      \fontsize{\bnSubtitleSize pt}{\bnSubtitleSize pt}\selectfont
      \textcolor{bnMuted}{\bnSubtitle}
    }
    %
    \fi
  \end{tabular}
  %
  \hfill
  \null
\end{minipage}%
\end{document}
